Function basicAjax() gets data with AJAX in a thread, sorts it and returns the first 5 elements of the array. We run this function in 2 different threads providing different URLs, and then join responses from these 2 threads. 


function basicAjax(config) {
    var result = vkhttp(config); // http request, part of vkThread plugin (see API) 
	
    function compare(a,b){
      return b.stargazers_count - a.stargazers_count;
    }

    return JSON.parse(result).sort(compare).slice(0,5);
 }



 var param1 = {
    fn: examples.basicAjax,
    args: [ {url: 'https://api.github.com/users/angular/repos'} ]
};
var param2 = {
    fn: examples.basicAjax,
    args: [ {url: 'https://api.github.com/users/vkiryukhin/repos'} ]
}

  vkThread.execAll([param1,param2]).then(
      function (data) {
        $scope.repos = data[0].concat(data[1]);
